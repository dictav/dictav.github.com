Dojo で DOM 要素を点滅させたいなーって思った。

「一発でできるのがあるだろ！」と気楽に探してみたんだけど、意外に見当たらない。Dojo のページの API 見れないのも痛い。

それでもぼちぼち探してみて、自分で作ってみたので公開しておこう。

dojo.require("dojox.fx")
function blink(_arg){
  color1 = _arg.start || 'red'
  color2 = _arg.end || 'white'
  duration = _arg.duration || 1500
  times = _arg.times || 10
  dt = duration / 10 / 2

  var fout = dojo.fadeOut({node:_arg.node, duration:dt,
                           properties:{backgroundColor:{
                             start:color2, end:color1 } 
                           }})
  var  fin = dojo.fadeIn({node:_arg.node, duration:dt,
                           properties:{backgroundColor:{
                             start:color2, end:color1 } 
                           }})
  _blink = [fout,fin]
  blinks = []
  for(i=0; i < times; i++){
    blinks = blinks.concat(_blink)
  }
  return dojox.fx.chain(blinks)
}
